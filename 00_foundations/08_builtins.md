# ⚙️ Встроенные функции Python

Встроенные функции — уже готовы, не нужен `import`.

---

## Вывод и ввод

```python
print("Привет")                    # вывод на экран
print("Имя:", name, "Возраст:", age)
print(f"Клиент: {name}")          # f-строка — лучший способ

name = input("Введите имя: ")     # ввод с клавиатуры (возвращает str)
age = int(input("Возраст: "))     # сразу преобразуем в int
```

## Типы и преобразование

```python
int("35")        # → 35        строка → целое
float("3.14")    # → 3.14      строка → дробное
str(35)          # → "35"      число → строка
bool(0)          # → False     0 и "" → False, всё остальное → True

type(35)         # <class 'int'>
isinstance(35, int)   # True   — проверить тип
```

## Числа

```python
abs(-5)          # 5      — модуль числа
round(3.567, 2)  # 3.57   — округление до 2 знаков
round(3.5)       # 4      — до целого

max(1, 5, 3)     # 5
min(1, 5, 3)     # 1
max([15, 34, 8]) # 34     — работает и со списком
sum([15, 34, 8]) # 57
pow(2, 10)       # 1024   — 2^10
```

## Списки и коллекции

```python
len([1, 2, 3])        # 3    — длина
len("Иванов")         # 6    — длина строки

sorted([3, 1, 2])             # [1, 2, 3] — новый отсортированный список
sorted([3, 1, 2], reverse=True)  # [3, 2, 1]

list(range(5))        # [0, 1, 2, 3, 4]
list(range(1, 6))     # [1, 2, 3, 4, 5]
list(range(0, 10, 2)) # [0, 2, 4, 6, 8] — с шагом 2

enumerate(["а", "б", "в"])  # (0,"а"), (1,"б"), (2,"в")
zip([1, 2], ["а", "б"])     # (1,"а"), (2,"б")
```

## Строки

```python
s = "  Иванов Алексей  "

s.strip()            # "Иванов Алексей"     — убрать пробелы
s.upper()            # "  ИВАНОВ АЛЕКСЕЙ  " — заглавные
s.lower()            # "  иванов алексей  " — строчные
s.title()            # "  Иванов Алексей  " — каждое слово с заглавной
s.split()            # ["Иванов", "Алексей"] — разбить по пробелу
s.split(",")         # разбить по запятой
", ".join(["а","б","в"])   # "а, б, в" — склеить список в строку
s.replace("Иванов", "Петров")
s.startswith("  И")  # True
s.endswith("ей  ")   # True
"Алексей" in s       # True
```

## Файлы

```python
open("file.txt", "r")   # открыть для чтения
open("file.txt", "w")   # открыть для записи (очистит файл!)
open("file.txt", "a")   # открыть для добавления в конец

# Всегда используйте with — файл закроется автоматически
with open("clients.txt", "r", encoding="utf-8") as f:
    content = f.read()      # всё содержимое строкой
    lines = f.readlines()   # список строк
```
